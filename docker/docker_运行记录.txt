docker start centos7

启动ssh
> /usr/sbin/sshd -D

报错执行以下命名
ssh-keygen -t rsa -f /etc/ssh/ssh_host_rsa_key -N ""
ssh-keygen -t ecdsa -f /etc/ssh/ssh_host_ecdsa_key -N ""
ssh-keygen -t ed25519 -f /etc/ssh/ssh_host_ed25519_key -N ""

vim /etc/ssh/sshd_config
